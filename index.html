let products = []; let selectedProducts = []; // Load products.json async function loadProducts() { const res = await fetch("products.json"); products = await res.json(); renderProducts(products); } // Render product cards function renderProducts(list) { const container = document.getElementById("productsContainer"); container.innerHTML = ""; list.forEach((product, index) => { const card = document.createElement("div"); card.className = "product-card"; card.innerHTML = `

${product.name}

${product.brand}

${product.category}

`; container.appendChild(card); }); } // Select / unselect product function toggleProduct(index) { const product = products[index]; const exists = selectedProducts.find(p => p.name === product.name); if (exists) { selectedProducts = selectedProducts.filter(p => p.name !== product.name); } else { selectedProducts.push(product); } updateSelectedUI(); renderProducts(products); } // Update selected products UI function updateSelectedUI() { const list = document.getElementById("selectedProductsList"); list.innerHTML = ""; selectedProducts.forEach((p) => { const item = document.createElement("div"); item.className = "selected-item"; item.innerHTML = `

${p.name}

`; list.appendChild(item); }); // save locally localStorage.setItem("selectedProducts", JSON.stringify(selectedProducts)); } // Remove selected product function removeSelected(name) { selectedProducts = selectedProducts.filter(p => p.name !== name); updateSelectedUI(); renderProducts(products); } // Generate routine (FAKE AI RESPONSE for now) function generateRoutine() { const chat = document.getElementById("chatWindow"); if (selectedProducts.length === 0) { chat.innerHTML += `

AI: Please select products first.

`; return; } let routine = "Here is your personalized routine:\n\n"; selectedProducts.forEach((p, i) => { routine += `${i + 1}. ${p.name} - ${p.category}\n`; }); routine += "\nUse these daily for best results."; chat.innerHTML += `

AI Routine:
${routine.replace(/\n/g, "
")}

`; } // Chat system (simple follow-up) document.getElementById("chatForm").addEventListener("submit", (e) => { e.preventDefault(); const input = document.getElementById("userInput"); const chat = document.getElementById("chatWindow"); const userText = input.value; chat.innerHTML += `

You: ${userText}

`; chat.innerHTML += `

AI: I can help you refine your routine based on your selected products.

`; input.value = ""; }); // Generate button document.getElementById("generateRoutine").addEventListener("click", generateRoutine); // Category filter document.getElementById("categoryFilter").addEventListener("change", (e) => { const filtered = products.filter(p => p.category === e.target.value); renderProducts(filtered); }); // Load saved data function loadSaved() { const saved = localStorage.getItem("selectedProducts"); if (saved) { selectedProducts = JSON.parse(saved); updateSelectedUI(); } } // Init loadProducts(); loadSaved();
